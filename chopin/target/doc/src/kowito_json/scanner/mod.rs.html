7pub struct Scanner<'a> {
8    input: &'a [u8],
9}
10
11impl<'a> Scanner<'a> {
12    #[inline(always)]
13    pub fn new(input: &'a [u8]) -> Self {
14        Self { input }
15    }
16
17    /// Scan the entire input and build the tape inside the provided slice.
18    /// Employs Runtime JIT Dispatch to use the fastest instruction set available.
19    pub fn scan(&self, tape: &mut [u32]) -> usize {
20        #[cfg(target_arch = "x86_64")]
21        {
22            if std::is_x86_feature_detected!("avx2") && std::is_x86_feature_detected!("pclmulqdq") {
23                // Safety: Verified AVX2 and PCLMULQDQ are supported on this CPU at runtime
24                return unsafe { avx2::scan_avx2(self.input, tape) };
25            }
26        }
27
28        #[cfg(target_arch = "aarch64")]
29        {
30            // Note: std::arch::is_aarch64_feature_detected!("sve2") is currently unstable
31            // In a production Apple Silicon (M4+) environment, we would detect SVE/SVE2
32            // For now, NEON is always available on AArch64.
33            // SVE2 will be wired up here when `stdarch_aarch64_sve` stabilizes in rustc.
34
35            // Safety: NEON is guaranteed on aarch64.
36            return unsafe { neon::scan_neon(self.input, tape) };
37        }
38
39        // Fallback to pure portable SIMD (Fast, but without carry-less bitmanipulation)
40        // Reachable on non-aarch64 and non-x86_64_with_avx2 architectures
41        #[allow(unreachable_code)]
42        {
43            let generic_scanner = generic::Scanner::new(self.input);
44            generic_scanner.scan(tape)
45        }
46    }
47}

53    #[test]
54    fn test_dynamic_scan_simple_object() {
55        let json = b"{\"key\":\"value\"}";
56        let scanner = Scanner::new(json);
57        let mut tape = vec![0; 10];
58        let count = scanner.scan(&mut tape);
59
60        assert_eq!(count, 7);
61        assert_eq!(&tape[..count], &[0, 1, 5, 6, 7, 13, 14]);
62    }
