main.rs 
use chopin_core::{App, ApiResponse, get, Json};
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize)]
struct User {
    id: u32,
    name: String,
    email: String,
}

async fn hello() -> &'static str {
    "Hello, World!"
}

async fn get_user() -> ApiResponse<User> {
    ApiResponse::success(User {
        id: 1,
        name: "Alice".to_string(),
        email: "alice@example.com".to_string(),
    })
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let app = App::new().await?
        .route("/", get(hello))
        .route("/users/:id", get(get_user));
    
    app.run().await?;
    Ok(())
}

use chopin_core::{routing::get, Router, Json};
use chopin_core::extractors::{AuthUser, AuthUserWithRole};
use chopin_core::models::user::Role;
use chopin_core::response::ApiResponse;
use serde_json::json;

// Any authenticated user can access this
async fn profile(user: AuthUser) -> Json<ApiResponse<serde_json::Value>> {
    Json(ApiResponse::success(json!({
        "id": user.user_id,
        "email": user.email,
        "role": user.role
    })))
}

// Only admins and superusers can access this
async fn admin_dashboard(
    user: AuthUserWithRole<{ Role::Admin as u8 }>,
) -> Json<ApiResponse<&'static str>> {
    Json(ApiResponse::success("Welcome, admin!"))
}

fn routes() -> Router {
    Router::new()
        .route("/profile", get(profile))
        .route("/admin/dashboard", get(admin_dashboard))
}

database.rs 
use chopin_core::{extract::State, routing::get, Router, Json};
use chopin_core::response::ApiResponse;
use sea_orm::{EntityTrait, ColumnTrait, QueryFilter};

#[derive(Clone)]
struct AppState {
    db: sea_orm::DatabaseConnection,
}

async fn list_posts(
    State(state): State<AppState>,
) -> Json<ApiResponse<Vec<PostResponse>>> {
    let posts = Post::find()
        .filter(post::Column::Published.eq(true))
        .all(&state.db)
        .await
        .unwrap_or_default();
    
    let responses = posts.into_iter()
        .map(PostResponse::from)
        .collect();
    
    Json(ApiResponse::success(responses))
}

fn routes(state: AppState) -> Router {
    Router::new()
        .route("/posts", get(list_posts))
        .with_state(state)
}
